Multi-Dimensional Timeseries Analysis

4 comments Webdesign

Portion of a larger code Kriging expected bias produced by a Kalman Filter. Directories contain infomation based on Kriging paramaters with filenames with UTC time.

Ran different versions of this code in seperate terminals to create a database based on order of anisotropic detrending and correlation model for Kriging Designed to be run with one setting combination per terminal 

                                
                                clear
                                clc
                                
                                projPhase = 'KrigingData'; 
                                PredCorrGR = 'ExpectedBias';
                                KrigOutputs = 'ExpCorr_2ndReg';
                                
                                
                                % %Range of storms within a list 
                                % %that meet the criteria of the investigation>
                                for storm=3:109
                                    
                                    clearvars -except storm projPhase PredCorrGR KrigOutputs 
                                    
                                    StormIndex=num2str(storm);
                                    
                                    InputDir=fullfile(projPhase,[PredCorrGR,StormIndex]);
                                    OutputDir=fullfile(projPhase,[KrigOutputs,StormIndex]);
                                    mkdir(OutputDir)
                                
                                
                                    
                                    % %%Storm history, times UTC>>
                                    filebeg  = ['UTCindex',storm];
                                    STCVname=[filebeg,'SC24.txt'];
                                    STCVID=fopen(fullfile('storm',STCVname),'r');
                                    timnam = 1;
                                    while(~feof(STCVID));
                                        
                                        InputText =textscan(STCVID, '%s',1,'delimiter', ',');
                                        TimeCell{timnam,1} = InputText{1};
                                        timnam = timnam+1;
                                        
                                    end
                                    
                                    fclose(STCVID);
                                    % %%<<
                                
                                
                                    
                                    % %%%Range of available time depending on storm>>>
                                    for u = 1:(timnam-2)
                                        
                                        fprintf('Kriging \n')
                                        disp(u)
                                        
                                        clearvars ColMad C RowMad R ffMad ff S Y
                                        
                                        Cit = 1;
                                        Rit = 1;
                                        ffit = 1;
                                        
                                        filebeg  = cell2mat(TimeCell{u,1});%Time UTC
                                
                                
                                
                                
                                
                                        % %%%%Reading WRF domain Column = C, Row = R,
                                        % %%%%and KF predictor correction = FF >>>>
                                        filnCtr = ['GCHtrC',filebeg,'.txt'];
                                        filnRtr = ['GCHtrR',filebeg,'.txt'];
                                        filnFFtr= ['GCHtrFF',filebeg,'.txt'];
                                        
                                        CmIDtr = fopen(fullfile(InputDir,(filnCtr)), 'r');
                                        RmIDtr = fopen(fullfile(InputDir,(filnRtr)), 'r');
                                        FFmIDtr = fopen(fullfile(InputDir,(filnFFtr)), 'r');
                                        
                                        while(~feof(CmIDtr));
                                            
                                            ColMad =textscan(CmIDtr, '%s',1,'delimiter', '\t');
                                            sizeCMAD = size(ColMad{1});
                                            if sizeCMAD(1)>0;
                                                C{Cit,1} = ColMad{1};
                                                Cit = Cit+1;
                                            end
                                        end
                                        
                                        while(~feof(RmIDtr));
                                            
                                            RowMad =textscan(RmIDtr, '%s',1,'delimiter', '\t');
                                            sizeRMAD = size(RowMad{1});
                                            if sizeRMAD(1)>0;
                                                R{Rit,1} = RowMad{1};
                                                Rit = Rit+1;
                                            end
                                        end
                                        
                                        while(~feof(FFmIDtr));
                                            
                                            ffMad =textscan(FFmIDtr, '%s',1,'delimiter', '\t');
                                            sizeffMAD = size(ffMad{1});
                                            if sizeffMAD(1)>0;
                                                ff{ffit,1} = ffMad{1};
                                                ffit = ffit+1;
                                            end
                                        end
                                        
                                        fclose(CmIDtr);
                                        fclose(RmIDtr);
                                        fclose(FFmIDtr);
                                        
                                        sw = ffit-1;
                                        
                                        for q = 1:sw;
                                            Y(q,1) = str2num(cell2mat(ff{q,1}));%Inputs to the Kriging
                                            
                                            S(i,1) = str2num(cell2mat(C{i,1}));%Location of data
                                            S(i,2) = str2num(cell2mat(R{i,1}));
                                        end
                                        % %%%%<<<<
                                
                                
                                
                                
                                
                                        % %%%%%Settings and running Kriging Model
                                        theta = [10 10];
                                        %lower boundary for theta guesses
                                        lob = [1e-100 1e-100];
                                        %Upper boundary for theta guesses
                                        upb = [1000 1000];
                                       
                                        [dmodel, perf] = ...
                                            dacefit(S, Y, @regpoly2, @correxp, theta, lob, upb);
                                        
                                    
                                        X = gridsamp([0, 0;330 330], 330);
                                        [YX MSE] = predictor(X, dmodel);
                                        X1 = reshape(X(:,1),330, 330);
                                        X2 = reshape(X(:,2),330, 330);
                                        YX_1D = YX;  
                                        YX = reshape(YX, size(X1));
                                        % %%%%%<<<<<
                                
                                
                                
                                
                                
                                        % %%%%%%Optional 3-D plots>>>>>>
                                        % Settings_list=[filebeg,' ',KrigOutputs]
                                
                                        %figure(1),  mesh(X1, X2, YX);
                                        % % hold on,
                                        % % plot3(S(:,1),S(:,2),Y,'.k', 'MarkerSize',330)
                                        % % hold off
                                        % title(Settings_list)
                                        % xlabel('WRF Cell Location Row')
                                        % ylabel('WRF Cell Location Column')
                                        % zlabel('Kriged Expected Bias (m/s)')
                                        
                                
                                        % figure(2),  mesh(X1, X2, reshape(MSE, size(X1)))
                                        % xlabel('WRF Cell Location Row')
                                        % ylabel('WRF Cell Location Column')
                                        % zlabel('MSE (m/s)')
                                        % mseSettings_list=[filebeg, ' MSE ', KrigOutputs]
                                        % title(mseSettings_list)
                                        % %%%%%%<<<<<<
                                        
                                
                                
                                
                                
                                        % %%%%%%%Relevant files for analysis>>>>>>>
                                        yxn=[filebeg,'KrigedOut.txt'];
                                        IntMSE=[filebeg,'InterpMSE.txt'];
                                        locat=[filebeg,'InputLocat.txt'];
                                        orig=[filebeg,'InputPredCorr.txt'];
                                        
                                        
                                        dlmwrite(fullfile(OutputDir,yxn),YX_1D);
                                        dlmwrite(fullfile(OutputDir,IntMSE),MSE);
                                        dlmwrite(fullfile(OutputDir,locat),S);
                                        dlmwrite(fullfile(OutputDir,orig),Y);
                                        % %%%%%%%<<<<<<<
                                        
                                    end
                                    % %%%<<<
                                end
                                % %<
